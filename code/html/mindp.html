function [] = mindp()
%sample code to find the (p, d) of a prosumer that minimizes its objective

% price, which is known from the last iteration
lambda = 0;
newLambda = 2.5;
user = [0 5 20 15 0.015 0.038 -0.008 0.8;
           0 7 25 18 0.008 0.047 -0.014 0.5;
           0 10 30 25 0.011 0.056 -0.009 0.4];

userdataTable = array2table(user, 'VariableNames', {'plow', 'dlow', 'phigh', 'dhigh' ,'alpha1' , 'alpha2' ,'beta1', 'beta2'});


% other parameters
a = 100; I = 3;
k = 0;
tolerance = 0.0001;


while abs(newLambda - lambda) > tolerance
k = k + 1;
lambda = newLambda;
for i = 1:I

    % objective function x=[x(1);x(2)] = [p;d]

    %function x = solveoptimization(plow,dlow,phigh,dhigh,alpha1,alpha2,beta1,beta2, a, I, newLambda)
    x = solveoptimization(0, 5, 20, 15, 0.015, 0.038, -0.008, 0.8, a, I, newLambda);
    %x = solveoptimization(user1{1},user1{2},user1{3},user1{4},user1{5},user1{6},user1{7},user1{8}, a, I, newLambda);
    p1 = x(1);
    d1 = x(2);

    y = solveoptimization(0, 7, 25,18,0.008,0.047,-0.014, 0.5, a, I, newLambda);
    p2 = y(1);
    d2 = y(2);

    z = solveoptimization(0, 10, 30,25,0.011,0.056,-0.009, 0.4, a, I, newLambda);
    p3 = z(1);
    d3 = z(2);

    %output result
    bit1 = d1 - p1 + a*lambda;
    bit2 = d2 - p2 + a*lambda;
    bit3 = d3 - p3 + a*lambda;


end
bitSum =bit1 + bit2 + bit3;
newLambda = bitSum/(a*I);

end
x = transpose(x);
y = transpose(y);
z = transpose(z);
Energy_sharing_game = [x;y;z];
end
